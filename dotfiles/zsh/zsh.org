#+begin_src sh
zstyle :compinstall filename '/home/ax/.zshrc'
autoload -Uz compinit
compinit
#+end_src

* alias
=preview-tabbed= does not display text files when =VISUAL=emacsclient -nw=,
so we use =nvim= for now
#+begin_src sh
alias e='emacs -nw'
alias n="VISUAL='nvim' nnn -a -e"
alias nf='clear && fastfetch --config neofetch && echo'
alias l='eza -l --group-directories-first --icons'
alias lll='eza -al --group-directories-first --icons'
alias v=nvim
alias vim=nvim
#+end_src

* HOLD abbr
https://github.com/olets/zsh-abbr does run on OpenBSD, but throws errors,
e.g. =head: unknown option -- c=

Don't wanna bother with errors, else could just continue using fish, so we uninstalled it...

* prompt
testing https://github.com/sindresorhus/pure
** 1. add to fpath
#+begin_src sh
fpath+=($HOME/.zsh/pure)
#+end_src
** 2. set opts
#+begin_src sh
export PURE_PROMPT_SYMBOL='λ'
export PURE_CMD_MAX_EXEC_TIME=2
#+end_src
** 3. init
Initialize the prompt system and choose pure:
#+begin_src sh
autoload -U promptinit; promptinit
prompt pure
#+end_src

* fzf for searching history (=ctrl-r=)
#+begin_src sh
source <(fzf --zsh)
#+end_src
